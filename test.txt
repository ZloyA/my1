cd c:/ директория - зайти в папку
mkdir  имя - создать директорию
git clone путь к репозиторию в гитхабе
ls -la просмотр содержимого папки
cd  зайти в папку скопированную с гитбаша
git status  статус папки
start . открыть проводник в винде
vim names  имя файла - создание файла, заход внутрь файла
i  работа внутри файла
esc + :qw или zz выход из файла с сохранением
git add .   - добавить все файлы в папаке в гитхаб. если один то вместо точки имя файла
git commit -m "комментарий, что сделали"
git push  отправка на гитхаб
